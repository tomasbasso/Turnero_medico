    IMPORTS:
      import { NextRequest } from 'next/server'
      import { prisma } from '@/lib/prisma'
      import { PrismaClientKnownRequestError } from '@prisma/client/runtime/library'

    NOTA sobre el import de PrismaClientKnownRequestError: antes de escribir el archivo, verificar el path correcto en node_modules/@prisma/client/runtime/ para Prisma 6.19.3. Si el import desde '@prisma/client/runtime/library' genera error de módulo en runtime, probar '@prisma/client' directamente como fallback.

    Exportar función async POST(request: NextRequest):

    PASO 1 — PARSEAR BODY:
      const body = await request.json().catch(() => null)
      if (!body) return Response.json({ error: 'Cuerpo requerido' }, { status: 400 })

    PASO 2 — EXTRAER CAMPOS:
      Desestructurar del body: doctorId, date, time, patientName, patientDni, patientPhone, patientEmail, patientInsurance

    PASO 3 — VALIDAR CAMPOS REQUERIDOS (retornar 400 ante el primer fallo):
      - Si !patientName o !String(patientName).trim(): return 400 'El nombre es requerido'
      - Si !doctorId o isNaN(Number(doctorId)): return 400 'doctorId inválido'
      - Si !date o typeof date !== 'string' o !/^\d{4}-\d{2}-\d{2}$/.test(date): return 400 'Fecha inválida'
      - Si !time o typeof time !== 'string': return 400 'Hora inválida'

    PASO 4 — VALIDAR DNI (D-01):
      const dniClean = String(patientDni ?? '').replace(/[\s.]/g, '')
      if (!/^\d{7,8}$/.test(dniClean)): return Response.json({ error: 'El DNI debe tener 7 u 8 dígitos numéricos.' }, { status: 400 })

    PASO 5 — VALIDAR TELÉFONO (D-02):
      const phoneClean = String(patientPhone ?? '').replace(/\D/g, '')
      if (phoneClean.length !== 10): return Response.json({ error: 'El teléfono debe tener exactamente 10 dígitos.' }, { status: 400 })

    PASO 6 — VALIDAR EMAIL OPCIONAL (D-03):
      Si patientEmail y typeof patientEmail === 'string' y patientEmail.trim() !== '':
        if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(patientEmail.trim())):
          return Response.json({ error: 'Ingresá un email válido.' }, { status: 400 })

    PASO 7 — CONSTRUIR FECHA (sin sufijo Z para evitar timezone shift):
      const [year, month, day] = date.split('-').map(Number)
      const appointmentDate = new Date(year, month - 1, day)

    PASO 8 — CREAR APPOINTMENT en try/catch:
      try {
        const appointment = await prisma.appointment.create({
          data: {
            doctorId: Number(doctorId),
            date: appointmentDate,
            time: String(time),
            patientName: String(patientName).trim(),
            patientDni: dniClean,
            patientPhone: phoneClean,
            patientEmail: patientEmail?.trim() || null,
            patientInsurance: patientInsurance?.trim() || null,
            status: 'PENDING',
          },
        })
        return Response.json({ appointment }, { status: 201 })
      } catch (error) {
        if (error instanceof PrismaClientKnownRequestError && error.code === 'P2002') {
          return Response.json(
            { error: 'Este turno ya fue reservado. Elegí otro horario.' },
            { status: 409 }
          )
        }
        console.error('[POST /api/public/appointments]', error)
        return Response.json({ error: 'Error al crear el turno' }, { status: 500 })
      }

    REGLAS CRÍTICAS:
    - status siempre hardcodeado como 'PENDING' — no aceptar status del body (D-04: el único guard es el @@unique)
    - notes no se lee del body (no expuesto en el wizard público)
    - NO importar verifyToken, COOKIE_NAME ni requireAdmin
    - Respuesta exitosa es 201, no 200
    - Usar Response.json() — NO NextResponse.json()
    - El mensaje de error 409 es exactamente: 'Este turno ya fue reservado. Elegí otro horario.' (per UI-SPEC)
