    Create components/admin/AvailabilityEditor.tsx as a Client Component.

    'use client' at top.

    Imports: useState from 'react'; Loader2, Plus, X from 'lucide-react'; cn from '@/lib/utils'.

    Type definitions:
    - type TimeRange = { startTime: string; endTime: string; id?: number }
    - type DayRanges = Record<number, TimeRange[]>
    - type AvailabilityItem = { id: number; dayOfWeek: number; startTime: string; endTime: string }
    - Props: { doctorId: number; initialAvailabilities: AvailabilityItem[]; role: 'ADMIN' | 'RECEPTIONIST' }

    Constants:
    - DAY_NAMES: string[] = ['Domingo', 'Lunes', 'Martes', 'Miércoles', 'Jueves', 'Viernes', 'Sábado'] (index matches dayOfWeek 0=Sunday)
    - DAY_ORDER: number[] = [1, 2, 3, 4, 5, 6, 0] (display order: Lunes first, Domingo last — per UI-SPEC "Lunes to Domingo")

    State:
    - const [ranges, setRanges] = useState<DayRanges>(() => { const initial: DayRanges = { 0: [], 1: [], 2: [], 3: [], 4: [], 5: [], 6: [] }; for (const avail of initialAvailabilities) { initial[avail.dayOfWeek].push({ startTime: avail.startTime, endTime: avail.endTime, id: avail.id }) }; return initial })
    - const [loading, setLoading] = useState(false)
    - const [error, setError] = useState('')
    - const [success, setSuccess] = useState(false)

    Helpers:
    - addRange(day: number): setRanges(prev => ({ ...prev, [day]: [...prev[day], { startTime: '09:00', endTime: '17:00' }] }))
    - removeRange(day: number, index: number): setRanges(prev => ({ ...prev, [day]: prev[day].filter((_, i) => i !== index) }))
    - updateRange(day: number, index: number, field: 'startTime' | 'endTime', value: string): setRanges(prev => { const updated = [...prev[day]]; updated[index] = { ...updated[index], [field]: value }; return { ...prev, [day]: updated } })

    handleSave (async):
    - setLoading(true); setError(''); setSuccess(false)
    - Client-side validation: iterate all days and ranges — if any range has endTime <= startTime, setError('La hora de fin debe ser posterior a la hora de inicio.'); setLoading(false); return
    - Build payload: const availabilities = Object.entries(ranges).flatMap(([day, rs]) => rs.map(r => ({ dayOfWeek: Number(day), startTime: r.startTime, endTime: r.endTime })))
    - try: const res = await fetch('/api/admin/doctors/' + doctorId + '/availability', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ availabilities }) })
    - if (!res.ok): const d = await res.json(); setError(d.error ?? 'No se pudo guardar. Intentá nuevamente.'); return
    - setSuccess(true)
    - catch: setError('No se pudo guardar. Intentá nuevamente.')
    - finally: setLoading(false)

    JSX: div className="space-y-6"

    Day sections — map over DAY_ORDER:
    For each dayIndex in DAY_ORDER:
    - Section: div with key={dayIndex} className="bg-surface rounded-xl shadow-card p-4"
    - Day heading: p className="text-xs font-semibold uppercase tracking-wide text-text-secondary mb-3" showing DAY_NAMES[dayIndex]
    - Ranges list: if ranges[dayIndex].length === 0: p className="text-xs text-text-muted mb-2" showing "Sin disponibilidad"
    - For each range at index i in ranges[dayIndex]:
      - Row: div className="flex items-center gap-2 mb-2" with key={i}
      - startTime input: input type="time" value={range.startTime} onChange={e => updateRange(dayIndex, i, 'startTime', e.target.value)} disabled={role !== 'ADMIN'} className={cn("rounded-lg border border-border px-2 py-1.5 text-sm text-text-primary w-28 focus:outline-none focus:border-[rgba(13,148,136,0.4)] focus:ring-2 focus:ring-[rgba(13,148,136,0.2)]", role !== 'ADMIN' && "opacity-60 cursor-not-allowed")}
      - Separator: span className="text-text-secondary" showing "—"
      - endTime input: same as startTime with field 'endTime'
      - Remove button (ADMIN only): if role === 'ADMIN', button type="button" aria-label="Eliminar franja" onClick={() => removeRange(dayIndex, i)} className="text-error hover:text-error transition-colors p-1" showing X icon h-4 w-4

    - Per-row validation error: after each range row, if endTime <= startTime, show p className="text-xs text-error ml-0" showing "La hora de fin debe ser posterior a la hora de inicio."

    - "Agregar franja" button (ADMIN only): if role === 'ADMIN', button type="button" onClick={() => addRange(dayIndex)} className="flex items-center gap-1 text-sm text-accent hover:text-primary transition-colors mt-1" showing Plus h-4 w-4 then "+ Agregar franja"

Loading: Loader2 h-4 w-4 animate-spin mr-2; not loading: "Guardar disponibilidad".
